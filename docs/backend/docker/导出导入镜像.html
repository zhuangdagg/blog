构建镜像 ​

bash
docker build -t image-name image-path

导出镜像 ​

bash
docker save imageId > .

导入镜像 ​

bash
docker load < image-file.tar

为镜像打上 tag ​

bash

运行镜像 ​

bash
docker run -itd -p 宿主port:容器port name

# eg:  docker run -itd --name node_demo --mount source=myvol,target=/usr/src/app/cache -p 3001:3000 node-express-demo

multi-stage build ​

Dockerfile
# syntax=docker/dockerfile:1

ARG NODE_VERSION=18.0.0

FROM node:${NODE_VERSION}-alpine as base
WORKDIR /usr/src/app
EXPOSE 3000

FROM base as dev
RUN --mount=type=bind,source=package.json,target=package.json \
    --mount=type=bind,source=package-lock.json,target=package-lock.json \
    --mount=type=cache,target=/root/.npm \
    npm ci --include=dev
USER node
COPY . .
CMD npm run dev

FROM base as prod
RUN --mount=type=bind,source=package.json,target=package.json \
    --mount=type=bind,source=package-lock.json,target=package-lock.json \
    --mount=type=cache,target=/root/.npm \
    npm ci --omit=dev
USER node
COPY . .
CMD node src/index.js

FROM base as test
ENV NODE_ENV test
RUN --mount=type=bind,source=package.json,target=package.json \
    --mount=type=bind,source=package-lock.json,target=package-lock.json \
    --mount=type=cache,target=/root/.npm \
    npm ci --include=dev
USER node
COPY . .
RUN npm run test
bash
docker build -t node-docker-image-test --progress=plain --no-cache --target test .
